619 {

621  simxInt startTime;
622  simxInt off=0;
623  simxInt initDataLength=dataLength;
624  if (dataLength==0)
625  return(0);
626 #ifdef _WIN32

628  buff=(simxUChar*)MapViewOfFile(_mmfConn[clientID],FILE_MAP_ALL_ACCESS,0,0,_mmfSize[clientID]+20);
629  if (buff!=0)
630  {
631  if (buff[0]!=1)
632  {
633  UnmapViewOfFile(buff);
634  return(0);

636 
637  while (dataLength>0)
638  {
639  /* Wait for previous data to be gone: */
640  while (buff[5]!=0)
641  {
642  if (extApi_getTimeDiffInMs(startTime)>1000)
643  {
644  UnmapViewOfFile(buff);
645  return(0);

653  ((int*)(buff+6))[1]=20;
654  ((int*)(buff+6))[2]=initDataLength;
655  dataLength=0;
656  }
657  else
